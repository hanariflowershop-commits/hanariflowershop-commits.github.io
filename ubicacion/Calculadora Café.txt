<div style="max-width:400px;margin:auto;font-family:sans-serif;">
  <h3>Calculadora Preparación Café Sidra</h3>

  <label>Intensidad de Sabor (Suave,Medio y Fuerte):</label><br>
  <select id="intensidad" onchange="calcular()">
    <option value="18">Suave (1:18)</option>
    <option value="16" selected>Medio (1:16)</option>
    <option value="13">Fuerte (1:13)</option>
  </select><br><br>

  <label>Gramos de café o mililitros de agua:</label><br>
  <select id="modo" onchange="cambiarModo()">
    <option value="cafe">Gramos de café</option>
    <option value="agua">Mililitros de agua</option>
  </select><br><br>

  <div id="inputCafe">
    <label>Gramos de café (Una cuchara tiene 5gr y 7gr aprox):</label><br>
    <input type="number" id="cafe" value="15" onchange="calcular()"> g<br><br>
  </div>

  <div id="inputAgua" style="display:none;">
    <label>Mililitros de agua:</label><br>
    <input type="number" id="agua" value="240" onchange="calcular()"> ml<br><br>
  </div>

  <div id="resultado"></div>
</div>

<script>
function cambiarModo() {
  const modo = document.getElementById("modo").value;
  document.getElementById("inputCafe").style.display = modo === "cafe" ? "block" : "none";
  document.getElementById("inputAgua").style.display = modo === "agua" ? "block" : "none";
  calcular();
}

function calcular() {
  const intensidad = parseFloat(document.getElementById("intensidad").value);
  const modo = document.getElementById("modo").value;
  let resultado = "";

  if (modo === "cafe") {
    const gramosCafe = parseFloat(document.getElementById("cafe").value);
    const agua = gramosCafe * intensidad;
    resultado = `<strong>Usa ${agua.toFixed(0)} ml de agua</strong>`;
  } else {
    const mlAgua = parseFloat(document.getElementById("agua").value);
    const cafe = mlAgua / intensidad;
    resultado = `<strong>Usa ${cafe.toFixed(1)} g de café</strong>`;
  }

  document.getElementById("resultado").innerHTML = resultado;
}
</script>
